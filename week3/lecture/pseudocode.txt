
// Linear Search 

for each door from left to right
    if 50 is behind door
        Return true
Return false 

for i from 0 n-1
    if 50 is behind doors[i]
        Return true
Return false

// Binary Search

if no doors left
    Return false
if 50 is behind middle door
    Return true
Else If 50 < middle door 
    Search left half
Else If 50 > middle door
    Search right half
